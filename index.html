import sys import json import os import requests import psutil import threading from PyQt6.QtWidgets import (QApplication, QMainWindow, QWidget, QVBoxLayout, QHBoxLayout, QTextEdit, QLineEdit, QPushButton, QLabel) from PyQt6.QtCore import Qt, QTimer, pyqtSignal, QPropertyAnimation, QRect from PyQt6.QtGui import QColor, QPainter, QBrush # --- SKILL DOSYASI (Sistem Yetenekleri) --- class BilalSkills: @staticmethod def get_battery(): battery = psutil.sensors_battery() return f"Pil Seviyesi: %{battery.percent}" if battery else "Pil bilgisi alınamadı." @staticmethod def get_system_time(): from datetime import datetime return f"Şu anki saat: {datetime.now().strftime('%H:%M:%S')}" @staticmethod def handle_file(action, filename, content=None): if action == "save": with open(filename, 'w', encoding='utf-8') as f: f.write(content) return f"Dosya başarıyla kaydedildi: {filename}" return "Geçersiz dosya işlemi." # --- ANA UYGULAMA --- class BilalAI(QMainWindow): def __init__(self): super().__init__() self.api_key_file = "api_key.json" self.api_key = self.load_api_key() self.init_ui() # Ses simülasyonu için değişken (Yapay zeka konuşurken titreşim sağlar) self.voice_amplitude = 0 self.is_speaking = False def load_api_key(self): if os.path.exists(self.api_key_file): with open(self.api_key_file, 'r') as f: return json.load(f).get("key") return None def save_api_key(self, key): with open(self.api_key_file, 'w') as f: json.dump({"key": key}, f) self.api_key = key def init_ui(self): self.setWindowTitle("BilalAI - Advanced OS Assistant") self.setFixedSize(500, 700) self.setStyleSheet("background-color: #0B0E14; color: white;") # Uzay Siyahı layout = QVBoxLayout() # 1. Görsel Ses Efekti Alanı (Daire ve Tanecikler) self.visualizer = VoiceVisualizer(self) layout.addWidget(self.visualizer) # 2. Sohbet Geçmişi self.chat_area = QTextEdit() self.chat_area.setReadOnly(True) self.chat_area.setStyleSheet("background-color: #161B22; border: none; border-radius: 10px; padding: 10px; color: #E6EDF3;") layout.addWidget(self.chat_area) # 3. Giriş Alanı input_layout = QHBoxLayout() self.user_input = QLineEdit() self.user_input.setPlaceholderText("BilalAI'ye bir şeyler sor...") self.user_input.setStyleSheet("background-color: #161B22; border: 1px solid #30363D; border-radius: 5px; padding: 8px;") self.user_input.returnPressed.connect(self.process_chat) send_btn = QPushButton("Gönder") send_btn.setStyleSheet("background-color: #238636; color: white; border-radius: 5px; padding: 8px;") send_btn.clicked.connect(self.process_chat) input_layout.addWidget(self.user_input) input_layout.addWidget(send_btn) layout.addLayout(input_layout) central_widget = QWidget() central_widget.setLayout(layout) self.setCentralWidget(central_widget) # API Anahtarı yoksa sor if not self.api_key: self.chat_area.append("Sistem: Lütfen OpenRouter API anahtarınızı girin.") def process_chat(self): text = self.user_input.text().strip() if not text: return if not self.api_key: self.save_api_key(text) self.chat_area.append("Sistem: API Anahtarı kaydedildi. Artık konuşabiliriz!") self.user_input.clear() return self.chat_area.append(f"Siz: {text}") self.user_input.clear() # Skill Kontrolü (Basit komut ayrıştırma) if "saat" in text.lower(): res = BilalSkills.get_system_time() self.ai_respond(res) elif "pil" in text.lower(): res = BilalSkills.get_battery() self.ai_respond(res) else: threading.Thread(target=self.get_ai_response, args=(text,)).start() def get_ai_response(self, prompt): try: response = requests.post( url="https://openrouter.ai/api/v1/chat/completions", headers={ "Authorization": f"Bearer {self.api_key}", "Content-Type": "application/json" }, data=json.dumps({ "model": "openrouter/free", "messages": [{"role": "user", "content": prompt}] }) ) data = response.json() reply = data['choices'][0]['message']['content'] self.ai_respond(reply) except Exception as e: self.ai_respond("Hata: API bağlantısı kurulamadı.") def ai_respond(self, text): self.chat_area.append(f"BilalAI: {text}") # Ses görselleştirmesini tetikle self.start_voice_animation() def start_voice_animation(self): self.is_speaking = True QTimer.singleShot(3000, lambda: setattr(self, 'is_speaking', False)) # --- GÖRSELLEŞTİRME BİLEŞENİ --- class VoiceVisualizer(QWidget): def __init__(self, parent): super().__init__(parent) self.setMinimumHeight(150) self.timer = QTimer(self) self.timer.timeout.connect(self.update) self.timer.start(50) self.parent = parent self.phase = 0 def paintEvent(self, event): painter = QPainter(self) painter.setRenderHint(QPainter.RenderHint.Antialiasing) center_x = self.width() // 2 center_y = self.height() // 2 # Yapay zeka konuşuyorsa titreşim şiddetini artır amplitude = 20 if self.parent.is_speaking else 2 self.phase += 0.2 # Daire içindeki tanecikleri çiz painter.setBrush(QBrush(QColor(0, 150, 255, 150))) painter.setPen(Qt.PenStyle.NoPen) for i in range(8): import math angle = (i * 45) + (self.phase * 10) rad = math.radians(angle) offset = math.sin(self.phase + i) * amplitude x = center_x + int((40 + offset) * math.cos(rad)) y = center_y + int((40 + offset) * math.sin(rad)) painter.drawEllipse(x, y, 8, 8) # Merkez Daire painter.setBrush(QBrush(QColor(0, 100, 200, 200))) painter.drawEllipse(center_x-20, center_y-20, 40, 40) if __name__ == "__main__": app = QApplication(sys.argv) window = BilalAI() window.show() sys.exit(app.exec())
